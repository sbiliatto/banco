create table usuario (
    id_usuario INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome varchar(100) NOT NULL,
    email varchar(100) NOT NULL,
    senha varchar(100) NOT NULL,
    tipo INTEGER,
    data_nascimento date,
    pontos integer
)

create table filme (
    id_filme INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    titulo varchar(200),
    sinopse blob sub_type 1,
    genero varchar(50),
    duracao integer,
    classificacao varchar(10),
    data_lancamento date,
    trailer varchar(255)
)

create table sala (
    id_sala INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome varchar(50),
    qtd_assentos integer
)


CREATE TABLE assento_sala(
    id_assento_sala INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_sala INTEGER,
    assento VARCHAR(10),
    CONSTRAINT fk_sala FOREIGN KEY (id_sala)
        REFERENCES sala (id_sala)
)




create table sessao (
    id_sessao INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_filme integer,
    id_sala integer,
    status integer,
    data date,
    horario time,
    valor_assento decimal(10,2),

    constraint fk_sessao_filme
        foreign key (id_filme)
        references filme(id_filme),

    constraint fk_sessao_sala
        foreign key (id_sala)
        references sala(id_sala)
)

create table reserva (
    id_reserva INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_promocao integer,
    id_usuario integer,
    id_sessao integer,
    valortotal decimal(10,2),
    desconto decimal(10,2),
    status varchar(20),
    datareserva date,

    constraint fk_reserva_usuario
        foreign key (id_usuario)
        references usuario(id_usuario),

    constraint fk_reserva_sessao
        foreign key (id_sessao)
        references sessao(id_sessao)
)

DROP TABLE SESSAO 


create table reserva_assento (
    id_reserva_assento INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_reserva integer,
    id_assento_sessao integer,


    constraint fk_reserva_assento_reserva
        foreign key (id_reserva)
        references reserva(id_reserva),

    constraint fk_reserva_assento_assento
        foreign key (id_assento_sessao)
        references assento_sala(id_assento_sala)
)

CREATE TABLE promocao (
    id_promocao INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR(100),
    descricao VARCHAR(256),
    tipo VARCHAR(50),
    valor DECIMAL(10,2),
    data_inicio DATE,
    data_fim DATE,
    status INTEGER
)
